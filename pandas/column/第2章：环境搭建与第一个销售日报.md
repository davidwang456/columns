# 第2章：环境搭建与第一个销售日报

## 1. 项目背景

某连锁零售公司在全国有 12 家门店，每天各门店店长通过邮件上报一个 Excel 格式的销售明细表（包含销售时间、商品编码、商品名称、品类、数量、单价、成交总额、门店名称）。总部运营经理小周需要每天下午 5 点前汇总所有门店的数据，生成一份按"日期→门店→品类"三级维度的销售日报，发送给区域总监。

当前流程是：小周每天下午手动下载 12 封邮件的 Excel 附件，逐一打开，复制粘贴到一张大表里，再用 Excel 的数据透视表拉出汇总结果。这套流程有三个致命问题：

**痛点一：效率低下**。12 家门店数据从下载到汇总至少需要 40 分钟，一旦某家门店邮件迟到，整个日报就卡住。更糟糕的是营销活动日（双十一、618），门店数量翻倍到 24 家，小周需要加班到晚上 8 点才能出报告。

**痛点二：准确性无法保障**。手工复制粘贴过程中，经常出现少选某行、错位粘贴的失误。上个月就发生过一次：小周把 3 号店的咖啡品类数据粘贴到了 1 号店下面，导致区域总监误判 1 号店业绩暴增，做了错误的促销资源调配决策。

**痛点三：无法扩展**。当总部要求把分析粒度从"门店×品类"细化为"门店×品类×单品"时，Excel 透视表的性能已经撑不住了。而且整个流程高度依赖小周个人，他请假时没有任何其他人能接手。

痛点流程：

```
门店1 Excel  ──┐
门店2 Excel  ──┤
  ...         ──┤── 手工下载→复制粘贴→透视表 ──→ 日报 ✓(40min)
门店12 Excel ──┘        ↑
                   容易错位 | 依赖个人 | 无法扩展
```

实际上，这恰好是 pandas 的经典应用场景：**批量的、结构化的、需要自动化生成日报的数据处理任务**。本章将从零开始搭建 pandas 开发环境，并用一份模拟的门店销售数据，完成第一个完整的销售日报项目。

### 3.2 模拟数据生成

**目标**：生成 12 家门店的模拟销售 Excel 文件，模拟真实业务场景。

```python
# generate_store_data.py
import pandas as pd
import numpy as np
from pathlib import Path
from datetime import datetime, timedelta

np.random.seed(42)

stores = [f"门店-{i:02d}" for i in range(1, 13)]
products = {
    '美式咖啡': '咖啡', '拿铁': '咖啡', '抹茶拿铁': '茶饮',
    '芝士蛋糕': '烘焙', '牛角包': '烘焙', '三明治': '轻食',
    '鸡肉沙拉': '轻食', '矿泉水': '饮品'
}

# 模拟日期范围：2025年3月15日
date = datetime(2025, 3, 15)

# 创建输出目录
output_dir = Path('store_data')
output_dir.mkdir(exist_ok=True)

for store in stores:
    n_records = np.random.randint(80, 150)
    records = []
    for _ in range(n_records):
        product_name = np.random.choice(list(products.keys()))
        category = products[product_name]
        quantity = np.random.randint(1, 5)
        # 在不同时段模拟不同客单价
        hour = np.random.randint(8, 21)
        minute = np.random.randint(0, 60)
        sale_time = date.replace(hour=hour, minute=minute)

        unit_price_map = {'美式咖啡': 22, '拿铁': 28, '抹茶拿铁': 32,
                          '芝士蛋糕': 35, '牛角包': 15, '三明治': 25,
                          '鸡肉沙拉': 30, '矿泉水': 5}
        unit_price = unit_price_map[product_name]
        total = quantity * unit_price

        records.append({
            '销售时间': sale_time.strftime('%Y-%m-%d %H:%M'),
            '商品编码': f'P{list(products.keys()).index(product_name)+1:03d}',
            '商品名称': product_name,
            '品类': category,
            '数量': quantity,
            '单价': unit_price,
            '成交总额': total,
            '门店名称': store
        })

    df = pd.DataFrame(records)
    df = df.sort_values('销售时间')
    filename = output_dir / f'{store}_20250315.xlsx'
    df.to_excel(filename, index=False, sheet_name='销售明细')

print(f"已生成 {len(stores)} 个门店的销售数据文件到 store_data/ 目录")
```

### 3.3 主流程实现

**目标**：读取所有门店 Excel 文件，合并、汇总，生成日报 Excel。

```python
# daily_report.py
import pandas as pd
from pathlib import Path
from datetime import datetime
from openpyxl import load_workbook
from openpyxl.styles import Font, Alignment, PatternFill, Border, Side

# ===== 1. 参数配置 =====
DATA_DIR = Path('store_data')
OUTPUT_DIR = Path('reports')
OUTPUT_DIR.mkdir(exist_ok=True)
REPORT_DATE = '2025-03-15'

print(f"[{datetime.now():%H:%M:%S}] 开始生成日报，日期：{REPORT_DATE}")

# ===== 2. 读取所有门店文件 =====
def load_all_stores(data_dir: Path, report_date: str) -> pd.DataFrame:
    """读取指定日期所有门店的 Excel 文件并合并"""
    all_data = []
    pattern = f"*_{report_date.replace('-', '')}.xlsx"
    files = list(data_dir.glob(pattern))

    if not files:
        raise FileNotFoundError(f"未找到匹配 '{pattern}' 的销售数据文件，请检查日期或目录")

    print(f"  找到 {len(files)} 个门店文件，开始读取...")

    for f in files:
        store_name = f.stem.rsplit('_', 1)[0]  # 从文件名提取门店名
        df = pd.read_excel(f)
        # 如果门店名列缺失，从文件名补全
        if '门店名称' not in df.columns:
            df['门店名称'] = store_name
        all_data.append(df)
        print(f"    ✓ {store_name}: {len(df)} 条记录")

    combined = pd.concat(all_data, ignore_index=True)
    print(f"  合并完成：共 {len(combined)} 条销售记录，{combined['门店名称'].nunique()} 家门店")
    return combined

# ===== 3. 数据质量校验 =====
def validate_data(df: pd.DataFrame) -> None:
    """执行基本数据质量检查"""
    errors = []
    # 必要列检查
    required = ['销售时间', '商品名称', '品类', '数量', '成交总额', '门店名称']
    for col in required:
        if col not in df.columns:
            errors.append(f"缺失必要列：{col}")

    # 数值范围检查
    if '成交总额' in df.columns:
        neg_count = (df['成交总额'] < 0).sum()
        if neg_count > 0:
            errors.append(f"存在 {neg_count} 条负金额记录")

    # 门店数量检查
    if '门店名称' in df.columns:
        store_count = df['门店名称'].nunique()
        if store_count < 10:
            errors.append(f"门店数量过少（{store_count}），可能有文件缺失")

    if errors:
        print("  ⚠ 数据质量告警：")
        for e in errors:
            print(f"    - {e}")
    else:
        print("  ✓ 数据质量校验通过")

# ===== 4. 生成汇总报表 =====
def generate_summary(df: pd.DataFrame) -> dict:
    """生成多维度汇总结果"""
    # 按日期-门店-品类汇总
    df['销售日期'] = pd.to_datetime(df['销售时间']).dt.date.astype(str)

    # 门店 × 品类 交叉表
    store_category = pd.pivot_table(
        df, values='成交总额', index='门店名称',
        columns='品类', aggfunc='sum', fill_value=0
    )

    # 品类排名
    category_rank = (
        df.groupby('品类')['成交总额']
        .sum()
        .sort_values(ascending=False)
        .reset_index()
    )
    category_rank.columns = ['品类', '成交总额']
    category_rank['占比'] = (category_rank['成交总额'] / category_rank['成交总额'].sum() * 100).round(2)

    # 门店排名
    store_rank = (
        df.groupby('门店名称')['成交总额']
        .sum()
        .sort_values(ascending=False)
        .reset_index()
    )
    store_rank.columns = ['门店名称', '成交总额']

    # 热销单品 Top 10
    top_products = (
        df.groupby('商品名称')
        .agg(销量=('数量', 'sum'), 销售额=('成交总额', 'sum'))
        .sort_values('销售额', ascending=False)
        .head(10)
        .reset_index()
    )

    return {
        'store_category': store_category,
        'category_rank': category_rank,
        'store_rank': store_rank,
        'top_products': top_products
    }

# ===== 5. 导出带样式的 Excel 报告 =====
def export_report(df: pd.DataFrame, summaries: dict, report_date: str) -> None:
    """导出多 Sheet 带格式的 Excel 日报"""
    filename = OUTPUT_DIR / f"销售日报_{report_date.replace('-', '')}.xlsx"

    with pd.ExcelWriter(filename, engine='openpyxl') as writer:
        df.to_excel(writer, sheet_name='销售明细', index=False)
        summaries['store_category'].to_excel(writer, sheet_name='门店×品类')
        summaries['category_rank'].to_excel(writer, sheet_name='品类排名', index=False)
        summaries['store_rank'].to_excel(writer, sheet_name='门店排名', index=False)
        summaries['top_products'].to_excel(writer, sheet_name='热销Top10', index=False)

    # openpyxl 添加样式
    wb = load_workbook(filename)
    header_font = Font(bold=True, size=11)
    header_fill = PatternFill(start_color='4472C4', end_color='4472C4', fill_type='solid')
    header_font_white = Font(bold=True, size=11, color='FFFFFF')

    for sheet_name in wb.sheetnames:
        ws = wb[sheet_name]
        # 表头样式
        for cell in ws[1]:
            cell.font = header_font_white
            cell.fill = header_fill
            cell.alignment = Alignment(horizontal='center')
        # 自适应列宽
        for col in ws.columns:
            max_len = max(len(str(cell.value or '')) for cell in col)
            ws.column_dimensions[col[0].column_letter].width = min(max_len + 4, 50)

    wb.save(filename)
    print(f"  ✓ 日报已导出至 {filename}")

# ===== 主流程 =====
try:
    df = load_all_stores(DATA_DIR, REPORT_DATE)
    validate_data(df)
    summaries = generate_summary(df)

    print(f"\n===== 品类销售排名 =====")
    print(summaries['category_rank'].to_string(index=False))

    print(f"\n===== 门店 Top 5 =====")
    print(summaries['store_rank'].head().to_string(index=False))

    export_report(df, summaries, REPORT_DATE)
    print(f"\n[{datetime.now():%H:%M:%S}] 日报生成完成！")

except Exception as e:
    print(f"\n  ✗ 日报生成失败：{e}")
    raise
```

### 3.5 常见坑及解决方法

| 问题 | 原因 | 解决方法 |
|------|------|----------|
| `No module named 'pandas'` | 未激活虚拟环境或未安装 | 确认命令行有 `(sales_env)` 前缀，重跑 `pip install pandas` |
| `FileNotFoundError: No such file or directory` | 路径错误或数据文件未生成 | 先执行 `generate_store_data.py` 生成模拟数据 |
| Excel 打开报错"文件已损坏" | openpyxl 版本不兼容 | `pip install --upgrade openpyxl` |
| 数值显示为科学计数法 | Excel 默认格式 | 在 `to_excel` 后设置单元格格式为数字 |
| Windows 下 `\` 路径问题 | 混用反斜杠和正斜杠 | 统一使用 `pathlib.Path` 处理路径 |

## 4. 项目总结

### 4.1 优点 & 缺点

| 维度 | 本章方案 (pandas 脚本) | 手工 Excel 操作 | 商业 BI 工具（如 Tableau） |
|------|----------------------|----------------|--------------------------|
| 开发成本 | 一次编写，免费 | 每日人工成本 | 按年订阅，成本高 |
| 执行效率 | 数秒完成 | 30-40 分钟 | 数秒到数分钟 |
| 准确性 | 可测试可验证 | 易出人为失误 | 依赖数据源质量 |
| 定制化 | 完全灵活 | 受限于 Excel 功能 | 受限于 BI 平台功能 |
| 分享性 | 输出 Excel，对方无需 Python | 天然可分享 | 需要 BI 平台账号 |
| 可维护性 | 代码 + Git 版本控制 | 无版本控制 | 仪表盘配置可版本化 |

### 4.2 适用场景

- **适用场景**：
  1. 每日/每周固定格式的销售/运营数据汇总报表
  2. 需要从多个数据源（多个 Excel、CSV）合并生成单一报告
  3. 需要自动化、定时运行的报表任务（可进一步接入 cron/调度器）
  4. 数据量在 1 万～ 100 万行之间的结构化数据处理
  5. 需要将结果输出为 Excel 格式供非技术人员阅读的场景
- **不适用场景**：
  1. 实时流式数据监控——pandas 是批处理工具，不适合流计算
  2. 需要复杂交互式图表的仪表盘——建议配合 BI 工具使用

### 4.3 注意事项

- **虚拟环境管理**：多人协作时，确保 `requirements.txt` 及时更新，避免"在我机器上能跑"的问题；
- **文件路径**：生产环境建议使用绝对路径或环境变量配置数据目录，避免工作目录切换导致的找不到文件问题；
- **Excel 引擎**：`to_excel` 默认使用 `openpyxl`，写入大文件时可选 `xlsxwriter` 获得更好性能；
- **日期处理**：统一在读取阶段用 `parse_dates` 转换，避免后续反复转换。

### 4.4 常见踩坑经验

1. **"找不到模块"反复出现**：一位同事在 VS Code 终端和系统终端用了不同的 Python 环境——VS Code 里选了系统 Python 没选虚拟环境的解释器。解决方法：`Ctrl+Shift+P` → Python: Select Interpreter → 选择虚拟环境中的 Python。
2. **to_excel 覆盖同名 Sheet 静默失败**：使用 `pd.ExcelWriter` 时，如果写入模式是 `'w'`（默认），已有的同名 Sheet 会被静默覆盖。如果需要追加，改用 `mode='a'`。
3. **路径分隔符在 Windows 和 Linux 间不同**：一位同事把项目从 Windows 拷到服务器（Linux）后，硬编码的 `'data\\stores'` 路径失效。使用 `pathlib.Path` 的 `/` 运算符代替字符串拼接，可以跨平台兼容。
