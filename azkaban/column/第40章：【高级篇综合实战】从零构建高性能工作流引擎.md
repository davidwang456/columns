# 第40章：【高级篇综合实战】从零构建高性能工作流引擎

## 1. 项目背景

### 业务场景

某头部互联网公司的数据平台经过3年发展，Azkaban上运行着5000+个Flow、日执行次数超过20000次。随着规模增长，Azkaban的架构瓶颈逐渐暴露：

1. **单Web Server调度瓶颈**：虽然做了HA，但调度能力仍受限于单节点性能
2. **状态同步延迟**：Executor心跳更新延迟导致Flow分配不均匀
3. **缺乏任务优先级**：所有Flow平等争抢资源
4. **无法水平扩展调度能力**：增加Web Server不能提升调度吞吐量

CTO决定：基于Azkaban的核心思想，自研一套更适合超大规模场景的下一代调度引擎。

### 痛点放大

需要解决的问题：
- 如何让调度能力随着节点增加而线性增长？
- 如何实现真正的任务优先级和资源隔离？
- 如何支持10万+任务/小时的调度吞吐量？
- 如何保证99.99%的调度可靠性？

## 3. 项目实战

### 3.1 最小原型：NextGen Scheduler

```go
package main

import (
    "container/heap"
    "fmt"
    "sync"
    "time"
)

// ============ 核心数据结构 ============

// Task 表示一个可执行的任务
type Task struct {
    ID          string
    Name        string
    Priority    int           // 优先级（越小越高）
    DependsOn   []string      // 依赖的Task ID列表
    Status      TaskStatus
    SubmitTime  time.Time
    StartTime   time.Time
    EndTime     time.Time
    RetryCount  int
    MaxRetries  int
    Command     string
    Resources   ResourceRequirement
}

type TaskStatus int

const (
    PENDING TaskStatus = iota
    READY
    RUNNING
    SUCCEEDED
    FAILED
)

type ResourceRequirement struct {
    CPU    int // 核心数
    Memory int // MB
}

// PriorityQueue 优先级队列
type PriorityQueue []*Task

func (pq PriorityQueue) Len() int { return len(pq) }

func (pq PriorityQueue) Less(i, j int) bool {
    // 先按优先级，再按提交时间
    if pq[i].Priority != pq[j].Priority {
        return pq[i].Priority < pq[j].Priority
    }
    return pq[i].SubmitTime.Before(pq[j].SubmitTime)
}

func (pq PriorityQueue) Swap(i, j int) { pq[i], pq[j] = pq[j], pq[i] }

func (pq *PriorityQueue) Push(x interface{}) {
    *pq = append(*pq, x.(*Task))
}

func (pq *PriorityQueue) Pop() interface{} {
    old := *pq
    n := len(old)
    item := old[n-1]
    *pq = old[0 : n-1]
    return item
}

// ============ DAG调度器 ============

type DAGScheduler struct {
    mu       sync.Mutex
    tasks    map[string]*Task
    readyPQ  PriorityQueue   // 就绪任务队列
}

func NewDAGScheduler() *DAGScheduler {
    s := &DAGScheduler{
        tasks:   make(map[string]*Task),
        readyPQ: make(PriorityQueue, 0),
    }
    heap.Init(&s.readyPQ)
    return s
}

// SubmitDAG 提交一个DAG，解析依赖关系
func (s *DAGScheduler) SubmitDAG(tasks []*Task) {
    s.mu.Lock()
    defer s.mu.Unlock()

    // 注册所有任务
    for _, task := range tasks {
        s.tasks[task.ID] = task
        task.Status = PENDING
        task.SubmitTime = time.Now()
    }

    // 解析依赖，将无依赖的任务标记为READY
    for _, task := range tasks {
        if s.allDependenciesMet(task) {
            task.Status = READY
            heap.Push(&s.readyPQ, task)
            fmt.Printf("[SCHED] Task %s READY (priority=%d)\n", task.ID, task.Priority)
        }
    }
}

func (s *DAGScheduler) allDependenciesMet(task *Task) bool {
    for _, depID := range task.DependsOn {
        dep, exists := s.tasks[depID]
        if !exists || dep.Status != SUCCEEDED {
            return false
        }
    }
    return true
}

// GetNextTask 获取下一个可执行的任务（优先级最高的）
func (s *DAGScheduler) GetNextTask() *Task {
    s.mu.Lock()
    defer s.mu.Unlock()

    if s.readyPQ.Len() == 0 {
        return nil
    }
    return heap.Pop(&s.readyPQ).(*Task)
}

// OnTaskCompleted 任务完成回调，检查下游依赖
func (s *DAGScheduler) OnTaskCompleted(taskID string, success bool) {
    s.mu.Lock()
    defer s.mu.Unlock()

    task := s.tasks[taskID]
    if success {
        task.Status = SUCCEEDED
        fmt.Printf("[SCHED] Task %s SUCCEEDED\n", taskID)
    } else {
        if task.RetryCount < task.MaxRetries {
            task.RetryCount++
            task.Status = READY
            heap.Push(&s.readyPQ, task)
            fmt.Printf("[SCHED] Task %s RETRY (%d/%d)\n",
                taskID, task.RetryCount, task.MaxRetries)
            return
        }
        task.Status = FAILED
        fmt.Printf("[SCHED] Task %s FAILED\n", taskID)
    }

    // 检查所有依赖此任务的下游
    for _, t := range s.tasks {
        if t.Status == PENDING && s.allDependenciesMet(t) {
            t.Status = READY
            heap.Push(&s.readyPQ, t)
            fmt.Printf("[SCHED] Task %s READY (after %s completed)\n",
                t.ID, taskID)
        }
    }
}

// ============ Worker池 ============

type WorkerPool struct {
    scheduler    *DAGScheduler
    maxWorkers   int
    activeWorkers int
    mu           sync.Mutex
}

func NewWorkerPool(scheduler *DAGScheduler, maxWorkers int) *WorkerPool {
    return &WorkerPool{
        scheduler:  scheduler,
        maxWorkers: maxWorkers,
    }
}

func (wp *WorkerPool) Start() {
    fmt.Printf("[POOL] Starting with %d workers\n", wp.maxWorkers)
    
    for i := 0; i < wp.maxWorkers; i++ {
        go wp.worker(i)
    }
}

func (wp *WorkerPool) worker(id int) {
    for {
        task := wp.scheduler.GetNextTask()
        if task == nil {
            time.Sleep(100 * time.Millisecond)
            continue
        }

        wp.mu.Lock()
        wp.activeWorkers++
        fmt.Printf("[WORKER-%d] Executing task %s (active=%d)\n",
            id, task.ID, wp.activeWorkers)
        wp.mu.Unlock()

        // 模拟任务执行
        task.Status = RUNNING
        task.StartTime = time.Now()
        success := wp.executeTask(task)
        task.EndTime = time.Now()

        wp.mu.Lock()
        wp.activeWorkers--
        wp.mu.Unlock()

        wp.scheduler.OnTaskCompleted(task.ID, success)
    }
}

func (wp *WorkerPool) executeTask(task *Task) bool {
    // 模拟执行：随机成功/失败
    duration := time.Duration(500+time.Now().UnixNano()%2000) * time.Millisecond
    time.Sleep(duration)
    
    // 80%成功率
    return time.Now().UnixNano()%10 < 8
}

// ============ 主程序 ============

func main() {
    scheduler := NewDAGScheduler()

    // 定义一个示例DAG
    //     A
    //    / \
    //   B   C
    //    \ /
    //     D
    //     |
    //     E
    tasks := []*Task{
        {ID: "A", Name: "Data Export", Priority: 1, MaxRetries: 2,
            Resources: ResourceRequirement{CPU: 2, Memory: 512}},
        {ID: "B", Name: "Clean Left", Priority: 2, DependsOn: []string{"A"}, MaxRetries: 1,
            Resources: ResourceRequirement{CPU: 4, Memory: 1024}},
        {ID: "C", Name: "Clean Right", Priority: 2, DependsOn: []string{"A"}, MaxRetries: 1,
            Resources: ResourceRequirement{CPU: 4, Memory: 1024}},
        {ID: "D", Name: "Aggregate", Priority: 3, DependsOn: []string{"B", "C"}, MaxRetries: 2,
            Resources: ResourceRequirement{CPU: 2, Memory: 2048}},
        {ID: "E", Name: "Report", Priority: 5, DependsOn: []string{"D"}, MaxRetries: 0,
            Resources: ResourceRequirement{CPU: 1, Memory: 256}},
    }

    // 提交DAG
    scheduler.SubmitDAG(tasks)

    // 启动Worker池
    pool := NewWorkerPool(scheduler, 3) // 3个并发Worker
    pool.Start()

    // 等待完成
    time.Sleep(30 * time.Second)

    // 打印最终状态
    fmt.Println("\n=== DAG最终状态 ===")
    for _, task := range tasks {
        duration := "N/A"
        if !task.StartTime.IsZero() && !task.EndTime.IsZero() {
            duration = task.EndTime.Sub(task.StartTime).String()
        }
        fmt.Printf("  %s (%s): %v (retries=%d, duration=%s)\n",
            task.ID, task.Name, task.Status, task.RetryCount, duration)
    }
}
```

### 3.3 下一代引擎核心特性

| 特性 | 实现方式 | 预期效果 |
|------|---------|---------|
| 分布式调度分片 | 一致性哈希分配Flow到Scheduler | 调度能力线性扩展 |
| 优先级抢占 | 高优任务可以抢占低优Worker | 关键任务0延迟 |
| 事件驱动 | Kafka作为事件总线 | 调度层和执行层完全解耦 |
| 云原生动扩缩 | Worker工作在K8s Pod中 | 根据队列深度自动扩缩 |
| CQRS读写分离 | 写模型=事件溯源，读模型=物化视图 | 查询不影响调度性能 |

## 4. 项目总结

从Azkaban到自研引擎的演进路线：
1. **第一阶段**：在Azkaban上做深度定制（自定义JobType、自定义Servlet）
2. **第二阶段**：用Azkaban的核心思想，重构调度层（本原型）
3. **第三阶段**：完全自研，支持10万+任务/小时

2. 如果要支持"定时调度"（类似Cron），应该在架构的哪一层实现？如何与Kafka事件驱动模型结合？
