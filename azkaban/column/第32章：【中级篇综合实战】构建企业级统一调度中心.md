# 第32章：【中级篇综合实战】构建企业级统一调度中心

## 1. 项目背景

### 业务场景

某中大型互联网公司的数据团队经过两年发展，已经在Azkaban上积累了300+个Flow、50+个项目、覆盖了电商、广告、金融三大业务线的数据需求。但随之而来的是管理上的混乱：

- 三大业务线的调度各自为政，广告线凌晨0点密集触发100个Flow，电商线凌晨2点触发150个Flow，资源争抢严重
- 没有统一的监控视图，运维同学需要在多个Grafana大盘之间切换
- 告警规则分散在各项目里，同一个问题可能触发5条不同的告警
- 新业务接入需要手动创建项目、配置权限、设置调度——整套流程下来要半天

CTO要求：在一个月内将Azkaban从一个"任务调度工具"升级为"企业级统一调度中心"。

### 痛点放大

需要构建的是"平台的平台"——一个统一入口，纳管所有Azkaban项目和Flow，提供标准化接入、统一监控、统一告警和自动化运维能力。

## 2. 项目设计——剧本式交锋对话

**小胖**（望着白板上密密麻麻的项目关系图）：大师，300个Flow、50个项目——每次新来一个需求，我要花半天配置。能不能搞个"一键接入"？

**小白**：具体实施步骤呢？

**大师**：分四个阶段：
1. **标准化**：定义Flow模板、命名规范、配置规范
2. **自动化**：脚本化项目创建、调度配置、权限分配
3. **监控化**：统一Prometheus指标 + Grafana大盘

#### 步骤2：项目自动创建脚本

**目标**：一键创建新项目的完整配置。

```python
#!/usr/bin/env python3
# project_bootstrap.py —— 项目自动引导

import os
import shutil
import subprocess
from azkaban_client_v2 import AzkabanClient

class ProjectBootstrapper:
    """新项目快速接入工具"""
    
    def __init__(self, client: AzkabanClient):
        self.client = client
        self.template_dir = "./templates"
    
    def bootstrap(self, config):
        """
        config = {
            "project_name": "ads_etl",
            "template": "etl_template",
            "params": {
                "SOURCE_DB_HOST": "192.168.1.100",
                "SOURCE_DB_NAME": "ads_db",
                "SOURCE_TABLE": "ads_events",
                "ALERT_EMAIL": "ads-team@company.com",
                "SCHEDULE_CRON": "0 0 3 * * ?",
                "YARN_QUEUE": "ads_etl"
            }
        }
        """
        project_name = config["project_name"]
        template_name = config["template"]
        params = config["params"]
        
        print(f"=== 引导创建项目: {project_name} ===")
        
        # 1. 从模板生成项目文件
        self._generate_project_files(project_name, template_name, params)
        
        # 2. 创建Azkaban项目
        self.client.create_project(project_name, "Auto-generated by bootstrap")
        
        # 3. 打包上传
        zip_path = self._build_zip(project_name)
        self.client.upload_project(project_name, zip_path)
        
        # 4. 配置调度
        if params.get("SCHEDULE_CRON"):
            for flow_name in self._get_flow_names(project_name):
                self.client.schedule_cron(
                    project_name, flow_name,
                    params["SCHEDULE_CRON"]
                )
                print(f"  ✓ 调度已配置: {flow_name}")
        
        # 5. 配置权限
        owner = params.get("OWNER", "admin")
        self._setup_permissions(project_name, owner)
        
        # 6. 注册到监控系统
        self._register_monitoring(project_name, params)
        
        print(f"✓ 项目 {project_name} 创建完成！")
        print(f"  URL: http://azkaban.company.com/manager?project={project_name}")
        return True
    
    def _generate_project_files(self, project_name, template_name, params):
        """从模板生成项目文件，替换变量"""
        template_path = os.path.join(self.template_dir, template_name)
        output_path = f"/tmp/azkaban_gen/{project_name}"
        
        if os.path.exists(output_path):
            shutil.rmtree(output_path)
        shutil.copytree(template_path, output_path)
        
        # 递归替换所有文件中的模板变量
        for root, dirs, files in os.walk(output_path):
            for fname in files:
                filepath = os.path.join(root, fname)
                with open(filepath, 'r') as f:
                    content = f.read()
                
                # 替换变量
                for key, value in params.items():
                    content = content.replace(f"${{{key}}}", str(value))
                
                with open(filepath, 'w') as f:
                    f.write(content)
        
        print(f"  项目文件已生成: {output_path}")
    
    def _build_zip(self, project_name):
        """打包为zip"""
        zip_path = f"/tmp/{project_name}.zip"
        project_dir = f"/tmp/azkaban_gen/{project_name}"
        
        import zipfile
        with zipfile.ZipFile(zip_path, 'w', zipfile.ZIP_DEFLATED) as zf:
            for root, dirs, files in os.walk(project_dir):
                for f in files:
                    file_path = os.path.join(root, f)
                    arcname = os.path.relpath(file_path, project_dir)
                    zf.write(file_path, arcname)
        
        return zip_path
    
    def _get_flow_names(self, project_name):
        """获取项目中的所有Flow名"""
        project_dir = f"/tmp/azkaban_gen/{project_name}"
        flows = []
        for root, dirs, files in os.walk(project_dir):
            for f in files:
                if f.endswith('.flow'):
                    flows.append(f.replace('.flow', ''))
        return flows
    
    def _setup_permissions(self, project_name, owner):
        """配置项目权限"""
        try:
            # 通过API设置权限（取决于Azkaban版本）
            self.client._post(f"/manager?project={project_name}&ajax=addProjectUserPermission",
                            data={"user": owner, "permission": "WRITER"})
        except:
            print(f"  ⚠️  权限配置需手动完成")
    
    def _register_monitoring(self, project_name, params):
        """注册到监控系统"""
        import json
        monitor_config = {
            "project": project_name,
            "owner": params.get("OWNER", "unknown"),
            "sla_hours": params.get("SLA_HOURS", 4),
            "alert_email": params.get("ALERT_EMAIL"),
            "yarn_queue": params.get("YARN_QUEUE", "default")
        }
        
        with open(f"/opt/monitoring/registry/{project_name}.json", 'w') as f:
            json.dump(monitor_config, f, indent=2)

if __name__ == '__main__':
    client = AzkabanClient("http://localhost:8081", "admin", "admin")
    bootstrapper = ProjectBootstrapper(client)
    
    # 接入广告业务ETL
    bootstrapper.bootstrap({
        "project_name": "ads_etl",
        "template": "etl_template",
        "params": {
            "SOURCE_DB_HOST": "10.0.1.100",
            "SOURCE_DB_NAME": "ads_prod",
            "SOURCE_TABLE": "ad_impressions",
            "ALERT_EMAIL": "ads-team@company.com",
            "SCHEDULE_CRON": "0 0 3 * * ?",
            "YARN_QUEUE": "ads_etl",
            "OWNER": "ads_dev_lead",
            "SLA_HOURS": "3"
        }
    })
```

#### 步骤3：统一资源调度

**目标**：根据时间段和业务优先级智能分配Yarn队列资源。

```python
#!/usr/bin/env python3
# resource_allocator.py —— 资源分配器

import json
from datetime import datetime

class ResourceAllocator:
    """统一资源调度器"""
    
    # 时间段定义
    TIME_SLOTS = {
        "night_batch": {"start": 0, "end": 6},    # 凌晨批处理
        "morning": {"start": 6, "end": 9},         # 早晨报表
        "daytime": {"start": 9, "end": 18},        # 白天
        "evening": {"start": 18, "end": 24},       # 晚上
    }
    
    # 队列分配策略
    QUEUE_POLICY = {
        "night_batch": {
            "core_pipeline": {"queue": "etl", "capacity": 60},
            "ads": {"queue": "ads_etl", "capacity": 25},
            "finance": {"queue": "fin_etl", "capacity": 15},
        },
        "morning": {
            "core_pipeline": {"queue": "etl", "capacity": 70},
            "ads": {"queue": "ads_etl", "capacity": 20},
            "finance": {"queue": "fin_etl", "capacity": 10},
        },
    }
    
    @classmethod
    def get_queue_for_flow(cls, project, flow, submit_time=None):
        """根据提交时间和项目确定应使用的队列"""
        if submit_time is None:
            submit_time = datetime.now()
        
        hour = submit_time.hour
        
        # 确定时间段
        current_slot = "daytime"
        for slot_name, slot_range in cls.TIME_SLOTS.items():
            if slot_range["start"] <= hour < slot_range["end"]:
                current_slot = slot_name
                break
        
        # 查找队列
        for business, config in cls.QUEUE_POLICY.get(current_slot, {}).items():
            if project.startswith(business):
                return config["queue"]
        
        return "default"
    
    @classmethod
    def allocate_resources(cls, projects_flows):
        """为所有待提交的Flow分配资源和优先级"""
        now = datetime.now()
        allocations = []
        
        for pf in projects_flows:
            queue = cls.get_queue_for_flow(pf["project"], pf["flow"], now)
            allocations.append({
                **pf,
                "queue": queue,
                "submit_time": now.isoformat(),
                "time_slot": cls._get_time_slot(now.hour),
                "priority": cls._get_priority(pf["project"])
            })
        
        # 按优先级排序
        allocations.sort(key=lambda x: x["priority"])
        return allocations
    
    @classmethod
    def _get_time_slot(cls, hour):
        for slot_name, slot_range in cls.TIME_SLOTS.items():
            if slot_range["start"] <= hour < slot_range["end"]:
                return slot_name
        return "unknown"
    
    @classmethod
    def _get_priority(cls, project):
        """优先级：核心 > 金融 > 广告 > 其他"""
        if project.startswith("core"):
            return 1
        elif project.startswith("finance"):
            return 2
        elif project.startswith("ads"):
            return 3
        return 4
```

#### 步骤5：全局Dashboard后端

**目标**：聚合所有项目的统计信息。

```python
#!/usr/bin/env python3
# unified_dashboard_api.py —— 统一大盘API

from flask import Flask, jsonify
from azkaban_client_v2 import AzkabanClient
import pymysql

app = Flask(__name__)

def get_overview():
    """获取全局概览"""
    conn = pymysql.connect(host='prod-db', user='azkaban', password='xxx', database='azkaban')
    cursor = conn.cursor(pymysql.cursors.DictCursor)
    
    # 最近24小时统计
    cursor.execute("""
        SELECT 
            COUNT(*) total,
            SUM(CASE WHEN status='SUCCEEDED' THEN 1 ELSE 0 END) success,
            SUM(CASE WHEN status='FAILED' THEN 1 ELSE 0 END) failed,
            SUM(CASE WHEN status='RUNNING' THEN 1 ELSE 0 END) running,
            AVG(end_time - start_time)/1000/60 avg_duration_min
        FROM execution_flows
        WHERE start_time > UNIX_TIMESTAMP(DATE_SUB(NOW(), INTERVAL 24 HOUR)) * 1000
    """)
    stats = cursor.fetchone()
    conn.close()
    
    return {
        "total_executions_24h": stats["total"],
        "success_rate": f"{stats['success']/stats['total']*100:.1f}%" if stats["total"] > 0 else "N/A",
        "failed_count": stats["failed"],
        "running_count": stats["running"],
        "avg_duration_min": f"{stats['avg_duration_min']:.1f}"
    }

def get_business_line_stats():
    """各业务线统计"""
    conn = pymysql.connect(host='prod-db', user='azkaban', password='xxx', database='azkaban')
    cursor = conn.cursor(pymysql.cursors.DictCursor)
    
    business_lines = {
        "core_pipeline": "核心管道",
        "ads_": "广告",
        "finance_": "金融",
    }
    
    result = []
    for prefix, label in business_lines.items():
        cursor.execute("""
            SELECT 
                COUNT(*) total,
                SUM(CASE WHEN status='SUCCEEDED' THEN 1 ELSE 0 END) success
            FROM execution_flows
            WHERE start_time > UNIX_TIMESTAMP(DATE_SUB(NOW(), INTERVAL 24 HOUR)) * 1000
              AND project LIKE %s
        """, (f"{prefix}%",))
        
        stats = cursor.fetchone()
        result.append({
            "business": label,
            "total": stats["total"],
            "success_rate": f"{stats['success']/stats['total']*100:.1f}%" if stats["total"] > 0 else "N/A"
        })
    
    conn.close()
    return result

@app.route('/api/overview')
def api_overview():
    return jsonify(get_overview())

@app.route('/api/business')
def api_business():
    return jsonify(get_business_line_stats())

if __name__ == '__main__':
    app.run(port=5100)
```
